MY_PROJECT_ID=Footy
# Build container
docker build -t gcr.io/$MY_PROJECT_ID/xgb-pp-ms:latest .
docker push gcr.io/$MY_PROJECT_ID/xgb-pp-ms:latest

# Deploy a Docker image on Cloud Run
gcloud run deploy xgb-pp-ms \
 --image gcr.io/$MY_PROJECT_ID/xgb-pp-ms:latest \
 --region europe-west1 \
 --platform managed \
 --memory 512Mi \
